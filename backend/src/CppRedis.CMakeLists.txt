cmake_minimum_required(VERSION 3.2 FATAL_ERROR)
project(cpp-redis-download NONE)
include(ExternalProject)

      CppRedis

	  SOURCE_DIR "${CMAKE_BINARY_DIR}/cpp-redis-src"
      BINARY_DIR "${CMAKE_BINARY_DIR}/cpp-redis-build"
      GIT_TAG "4.3.1"
      UPDATE_COMMAND ""
      INSTALL_COMMAND "" )
